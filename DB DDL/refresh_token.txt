CREATE TABLE refresh_token (
  refresh_token_id BIGINT PRIMARY KEY AUTO_INCREMENT,
  user_id          VARCHAR(36) NOT NULL,
  token            VARCHAR(255) NOT NULL,        -- 해싱된 토큰 저장용 (길이 최적화)
  issued_at        DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  expired_at       DATETIME NOT NULL,
  revoked          BOOLEAN NOT NULL DEFAULT FALSE, -- 필드명 통일 (is_ 제거)
  revoked_at       DATETIME,
  
  -- 기기 식별 정보
  device_id        VARCHAR(128),                 -- 기기 고유 ID
  device_type      VARCHAR(20),                  -- WEB, IOS, ANDROID
  device_name      VARCHAR(255),                 -- 기기 별칭 (예: "내 아이폰", "Chrome 브라우저")
  user_agent       VARCHAR(512),                 -- 상세 로그용
  
  created_at       DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,

  -- 인덱스 전략
  UNIQUE KEY uk_token (token),                   -- 토큰 조회용
  KEY idx_user_device (user_id, device_id),      -- 💡 특정 유저의 기기별 토큰 청소용 (성능 핵심)
  KEY idx_expired_at (expired_at),               -- 만료된 토큰 일괄 삭제 배치용

  CONSTRAINT fk_refresh_user
    FOREIGN KEY (user_id) REFERENCES users(user_id) ON DELETE CASCADE
);
